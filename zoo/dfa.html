def run_automaton(input_str):
    state = 'a' # Initial state
    for char in input_str:
        if state == 'a':
            if char == '0':
                state = 'b'
            elif char == '1':
                state = 'a'
        elif state == 'b':
            if char == '0':
                state = 'b'
            elif char == '1':
                state = 'c'
        elif state == 'c':
            if char == '0':
                state = 'b'
            elif char == '1':
                state = 'a'
    return state in ['c'] # Accept state
